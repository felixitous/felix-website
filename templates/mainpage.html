Shortest SuperString

11 - 19 - 2014

A small script in built in Python which takes many subsequences of DNA and reconstructs the initial DNA sequence. This is done with a greedy algorithm which finds the biggest overlap, combines the two strings, and removes the initial two strings from the list. The script repeatedly does this until only one string is left which represents the initial DNA sequence. 
 								
# Calculates all the overlaps for all the strings
overlap_values = {}
def mostoverlaps(content):
	global overlap_values
	max_result = 0
	winner1 = None
	winner2 = None
	i = 0
	j = 0
	while (i < len(content)):
		while (j < len(content)):
			if (i < j):
				first = content[i]
				second = content[j]
				key = first + "|" + second
				tmp_result = overlap_values.get(key)
				if tmp_result is None:
					tmp_result = substringfinder2(first, second)
				overlap_values.update({key : tmp_result})
				if tmp_result > max_result:
					max_result = tmp_result
					winner1 = first
					winner2 = second
			j += 1
		i += 1
		j = 0
	combined = substring_edge(winner1, winner2, max_result)
	if combined is None:
		combined = substring_edge(winner2, winner1, max_result)
	content.remove(winner1)
	content.remove(winner2)
	content.append(combined)
	return content
